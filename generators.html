def D(x,w):
    x = linear(x,w[0])
    x = leaky_relu(x,0.2)
    x = linear(x,w[1])
    x = leaky_relu(x,0.2)
    x = linear(x,w[2])
    x = torch.sigmoid(x)
    return x               

# Generator
def G(x,w):
    x = linear(x,w[3])
    x = relu(x,0.2)
    x = linear(x,w[4])
    x = relu(x,0.2)
    x = linear(x,w[5])
    x = torch.tanh(x)
    return x

def D(x,w):
    x = relu(conv2d(x,w[0], stride=(2, 2), padding=(1, 1)))
    x = relu(conv2d(x,w[1], stride=(2, 2), padding=(1, 1)))
    x = x.view(x.size(0), 6272)
    x = linear(x,w[2])
    x = torch.sigmoid(x)
    return x

    x = torch.tanh(conv_transpose2d(x,w[5], stride=(2, 2), padding=(1, 1)))
